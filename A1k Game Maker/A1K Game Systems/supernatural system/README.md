## ğŸ”§ Integration

### Basic Spirit Usage

```javascript
import { Spirit } from './spirits/Spirit.js';
import { getSpirit } from './spirits/SpiritRegistry.js';
import { updateSpirit } from './spirits/SpiritController.js';
import { AllSpiritSprites } from './art/AllSpiritSprites.js';

// Create spirit
const spiritData = getSpirit('ember_fox');
const spirit = new Spirit(spiritData, playerId);
spirit.equip();

// Update (in game loop)
updateSpirit(spirit, dt, player, enemies, projectileSystem);

// Render
const sprites = new AllSpiritSprites();
sprites.renderSpirit(ctx, spirit.id, spirit.x, spirit.y, {
  animTime: spirit.animTime,
  color: spirit.color,
  secondaryColor: spirit.secondaryColor,
  glowColor: spirit.glowColor,
});

// Get bonuses
const bonuses = spirit.getBonuses(); // { atkMul: 0.05, luck: 5 }
```

### Ability System Usage

```javascript
import { getAbility } from './spirits/AbilityRegistry.js';
import { AllAbilityEffects } from './art/AllAbilityEffects.js';

const effects = new AllAbilityEffects();

// Activate ability
const ability = getAbility('divine_barrier');
effects.activateDivineBarrier(player.x, player.y);

// Update and render
effects.update(dt);
effects.render(ctx);
```
